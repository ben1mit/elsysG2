76 esp_err_t example_connect(void)
77 {
78  if (s_connect_event_group != NULL) {
79  return ESP_ERR_INVALID_STATE;
80  }
81  s_connect_event_group = xEventGroupCreate();
82  start();
83  ESP_ERROR_CHECK(esp_register_shutdown_handler(&stop));
84  ESP_LOGI(TAG, "Waiting for IP");
85  xEventGroupWaitBits(s_connect_event_group, CONNECTED_BITS, true, true, portMAX_DELAY);
86  ESP_LOGI(TAG, "Connected to %s", s_connection_name);
87  ESP_LOGI(TAG, "IPv4 address: " IPSTR, IP2STR(&s_ip_addr));
88 #ifdef CONFIG_EXAMPLE_CONNECT_IPV6
89  ESP_LOGI(TAG, "IPv6 address: " IPV6STR, IPV62STR(s_ipv6_addr));
90 #endif
91  return ESP_OK;
92 }

226  eth_mac_config_t mac_config = ETH_MAC_DEFAULT_CONFIG();
227  eth_phy_config_t phy_config = ETH_PHY_DEFAULT_CONFIG();

232  mac_config.smi_mdio_gpio_num = CONFIG_EXAMPLE_ETH_MDIO_GPIO;
233  s_mac = esp_eth_mac_new_esp32(&mac_config);
