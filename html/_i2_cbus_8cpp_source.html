108 esp_err_t I2C::writeBit(uint8_t devAddr, uint8_t regAddr, uint8_t bitNum, uint8_t data, int32_t timeout) {
109  uint8_t buffer;
110  esp_err_t err = readByte(devAddr, regAddr, &buffer, timeout);
111  if (err) return err;
112  buffer = data ? (buffer | (1 << bitNum)) : (buffer & ~(1 << bitNum));
113  return writeByte(devAddr, regAddr, buffer, timeout);
114 }

116 esp_err_t I2C::writeBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t data, int32_t timeout) {
117  uint8_t buffer;
118  esp_err_t err = readByte(devAddr, regAddr, &buffer, timeout);
119  if (err) return err;
120  uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);
121  data <<= (bitStart - length + 1);
122  data &= mask;
123  buffer &= ~mask;
124  buffer |= data;
125  return writeByte(devAddr, regAddr, buffer, timeout);
126 }

132 esp_err_t I2C::writeBytes(uint8_t devAddr, uint8_t regAddr, size_t length, const uint8_t *data, int32_t timeout) {
133  i2c_cmd_handle_t cmd = i2c_cmd_link_create();
134  i2c_master_start(cmd);
135  i2c_master_write_byte(cmd, (devAddr << 1) | I2C_MASTER_WRITE, I2C_MASTER_ACK_EN);
136  i2c_master_write_byte(cmd, regAddr, I2C_MASTER_ACK_EN);
137  i2c_master_write(cmd, (uint8_t*) data, length, I2C_MASTER_ACK_EN);
138  i2c_master_stop(cmd);
139  esp_err_t err = i2c_master_cmd_begin(port, cmd, (timeout < 0 ? ticksToWait : pdMS_TO_TICKS(timeout)));
140  i2c_cmd_link_delete(cmd);
141 #if defined CONFIG_I2CBUS_LOG_READWRITES
142  if (!err) {
143  char str[length*5+1];
144  for (size_t i = 0; i < length; i++)
145  sprintf(str+i*5, "0x%s%X ", (data[i] < 0x10 ? "0" : ""), data[i]);
146  I2CBUS_LOG_RW("[port:%d, slave:0x%X] Write %d bytes to register 0x%X, data: %s",
147  port, devAddr, length, regAddr, str);
148  }
149 #endif
150 #if defined CONFIG_I2CBUS_LOG_ERRORS
151 #if defined CONFIG_I2CBUS_LOG_READWRITES
152  else {
153 #else
154  if (err) {
155 #endif
156  I2CBUS_LOGE("[port:%d, slave:0x%X] Failed to write %d bytes to__ register 0x%X, error: 0x%X",
157  port, devAddr, length, regAddr, err);
158  }
159 #endif
160  return err;
161 }
